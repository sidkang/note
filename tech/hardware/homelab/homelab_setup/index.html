  credentials: true

  allowed_hosts:

[Unit]
Description=WebDAV server
After=network.target

[Service]
Type=simple
User=media
ExecStart=/usr/local/bin/webdav --config /home/media/webdav.yaml
Restart=on-failure

[Install]
WantedBy=multi-user.target
